  CREATE TABLE RECARGAS (
  IDRECARGA INTEGER PRIMARY KEY AUTOINCREMENT,
  IDUSUARIO INTEGER NOT NULL,

  );

  CREATE TABLE TARJETAS_METRO (
  IDTARJETA INTEGER PRIMARY KEY AUTOINCREMENT,
  IDUSUARIO INTEGER NOT NULL,
  NUMERO_TARJETA TEXT NOT NULL UNIQUE,
  SALDO DOUBLE DEFAULT 0,
  ACTIVA INTEGER DEFAULT 1,
  FOREIGN KEY(IDUSUARIO) REFERENCES USUARIOS(IDUSUARIO)
  );

  CREATE TABLE LOGIN_HISTORIAL (
  IDLOGIN INTEGER PRIMARY KEY AUTOINCREMENT,
  IDUSUARIO INTEGER NOT NULL,
  FECHA DATETIME DEFAULT (datetime('now','localtime')),
  EXITO INTEGER NOT NULL,
  FOREIGN KEY(IDUSUARIO) REFERENCES USUARIOS(IDUSUARIO)
  );
